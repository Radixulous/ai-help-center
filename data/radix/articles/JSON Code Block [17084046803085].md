```
curl --request POST \
  --url https://api.radix.com/auth \
  --header 'Content-Type: application/json' \
  --header 'x-radix-version: 1.0.0' \
  --data '{
  "client_id": "<client id>",
  "client_secret": "<client secret>"
}'
```
